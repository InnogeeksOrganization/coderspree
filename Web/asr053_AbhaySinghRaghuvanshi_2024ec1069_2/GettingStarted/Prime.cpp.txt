#include <bits/stdc++.h>
using namespace std;
bool isPrime(int n){
    if(n==2 || n==3)return true;
    if(n%2==0 || n%3==0)return false;
    for(int i=5;i<sqrt(n);i+=6){
        if(n%i==0)return false;
        if(n%(i+2)==0)return false;
    }
    return true;
}
int main(){
    int t;
    cin >> t;
   if(isPrime(t))cout<<"prime"<<endl;
   else cout<<"not prime"<<endl;
    
}
