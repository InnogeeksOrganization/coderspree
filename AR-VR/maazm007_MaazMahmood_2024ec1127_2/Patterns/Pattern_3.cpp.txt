#include <iostream>
using namespace std;

int main(int argc, char **argv){
    int n;
    //printf("Enter the number of rows : ");
    scanf("%d", &n);
    for (int i = n; i >= 1; i--)
    {
        for (int j = 1; j <= n; j++)
        {
            if (j == i || j > i)
            {
                printf("*\t");
            }
            else
            {
                printf("\t");
            }
        }
        printf("\n");
    }
}
