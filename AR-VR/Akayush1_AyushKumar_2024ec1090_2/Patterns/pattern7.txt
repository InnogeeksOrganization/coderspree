#include <iostream>
using namespace std;

int main(int argc, char **argv){

  int i,j,n;
  scanf("%d",&n);
  for(i=1;i<=n;i++)
  {
      for(j=1;j<=i;j++)
      {
          if(i==j)
          {
              printf("*\t");
          }
          else\
          {
              printf("\t");
          }
      }
      printf("\n");
  }
  return 0;
    
}
